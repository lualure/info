  local n1=1
  local fmt = "%5s\t%5s\t%20s %20s\n"
  str.say(fmt, "n","same?","one","two")
  str.say(fmt, "-----","-----","-----","-----")
  while n1 < 1.5 do
    local i,j = s.create(), s.create()
    for _=1,100 do
      local val = r.r()^0.5
      s.update(i, val)
      s.update(j, val*n1) 
    end
    str.say("%5.3f\t%5s\t%20s\t%20s\n",
            n1,s.cliffsDelta(i,j), 
            tostring(tiles.tiles(i._all,0.25)), 
            tostring(tiles.tiles(j._all,0.25)))
    n1 = n1 *1.01
  end end
