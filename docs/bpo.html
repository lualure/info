local function surprises(t)
  local got,want,inc
  for _,r in pairs(t.rows) do
    if r.xpect then
      got = row.dominate(r,t)
      want = r.xpect
      r.xpect   = nil
      inc = want-got end end
  return {want=want,got=got,inc=inc/#t.rows}
end
